local fol = script.Parent
local TweenService = game:GetService("TweenService")
local Debris = game:GetService("Debris")

-- CONFIG
local DROP_CENTER1 = Vector3.new(-111.5, 63.196, 154)
local DROP_CENTER2 = Vector3.new(-11.5, 2.597, 73)
local DROP_SIZE1 = Vector3.new(1, 4, 7)
local DROP_SIZE2 = Vector3.new(174, 1.2, 242)

local BEAM_DURATION = 6          -- how long each beam/parts live
local COLOR_STEP = 0.03          -- color step (smaller = smoother)
local MOVE_SPEED = 0.4
local MOVE_INTERVAL = 1.2
local SPAWN_DELAY = 1
local MAX_BEAMS = 1           -- safe cap
local BEAM_THICKNESS = 2

-- helper: safe random position in a rectangle (keeps Y fixed)
local function getRandomDropPosition(center, size)
    local minX, maxX = center.X - size.X/2, center.X + size.X/2
    local minZ, maxZ = center.Z - size.Z/2, center.Z + size.Z/2
    return Vector3.new(
    math.random() * (maxX - minX) + minX,
    center.Y,
    math.random() * (maxZ - minZ) + minZ
    )
end

local function rainbowColor(h)
    return Color3.fromHSV(h % 1, 1, 1)
end

-- Creates a temporary disco part (returns part)
local function createTempPart(center, size)
    local p = Instance.new("Part")
    p.Anchored = true
    p.CanCollide = false
    p.Material = Enum.Material.Neon
    p.Shape = Enum.PartType.Ball
    p.Transparency = 1
    p.Size = Vector3.new(2,2,2)
    p.Position = getRandomDropPosition(center, size)
    p.Parent = fol
    
    -- point light
    local light = Instance.new("PointLight")
    light.Brightness = 6
    light.Range = 25
    light.Parent = p
    
    -- particle emitter (attached to part so Debris cleans it automatically)
    local emitter = Instance.new("ParticleEmitter")
    emitter.Texture = "rbxassetid://619439195"
    emitter.Rate = 35
    emitter.Lifetime = NumberRange.new(0.3, 0.6)
    emitter.Speed = NumberRange.new(0.5, 1.5)
    emitter.Size = NumberSequence.new({NumberSequenceKeypoint.new(0,0.4), NumberSequenceKeypoint.new(1,0)})
    emitter.LightEmission = 1
    emitter.Parent = p
    
    -- schedule automatic cleanup via Debris (safest)
    Debris:AddItem(p, BEAM_DURATION + 1)
    
    return p, light, emitter
end

-- Move part smoothly a few times while it exists, but stop when it is removed.
-- We animate movement using a finite number of steps rather than an infinite loop.
local function animateMovement(part, center, size)
    -- number of moves during lifetime
    local moves = math.max(1, math.floor((BEAM_DURATION)/math.max(0.0001, MOVE_INTERVAL)))
    task.spawn(function()
        for i = 1, moves do
            if not part or not part.Parent then break end
            local newPos = getRandomDropPosition(center, size)
            local tween = TweenService:Create(part, TweenInfo.new(MOVE_SPEED, Enum.EasingStyle.Sine, Enum.EasingDirection.InOut), {Position = newPos})
            tween:Play()
            -- wait smaller increments so if the part is removed we break quickly
            local waited = 0
            while waited < MOVE_INTERVAL do
                if not part or not part.Parent then break end
                task.wait(0.1)
                waited = waited + 0.1
            end
            if not part or not part.Parent then break end
        end
    end)
end

-- Animate color for a limited duration (BEAM_DURATION). This uses a finite loop.
local function animateColorFinite(targets)
    -- targets: list of {object, field} pairs. For Beam we set .Color = ColorSequence.new(c)
    local steps = math.max(1, math.floor(BEAM_DURATION / COLOR_STEP))
    task.spawn(function()
        for i = 0, steps do
            local hue = (i / steps)
            local c = rainbowColor(hue)
            for _, t in ipairs(targets) do
                local obj, kind = t[1], t[2]
                if not obj or not obj.Parent then
                    -- skip missing objects
                else
                    if kind == "partColor" then
                        obj.Color = c
                    elseif kind == "lightColor" then
                        obj.Color = c
                    elseif kind == "beamColor" then
                        -- beam.Color expects a ColorSequence
                        if obj and obj.Parent then
                            obj.Color = ColorSequence.new(c)
                        end
                    elseif kind == "emitterColor" then
                        if obj and obj.Parent then
                            obj.Color = ColorSequence.new(c)
                        end
                    end
                end
            end
            task.wait(COLOR_STEP)
        end
    end)
end

-- Create a beam connecting two temporary parts, animate for BEAM_DURATION and ensure cleanup.
local function spawnBeamPair(center1, size1, center2, size2)
    -- create two temporary parts
    local p1, light1, emitter1 = createTempPart(center1, size1)
    local p2, light2, emitter2 = createTempPart(center2, size2)
    
    -- attachments (parented to parts so they are cleaned with parts)
    local a1 = Instance.new("Attachment")
    a1.Parent = p1
    local a2 = Instance.new("Attachment")
    a2.Parent = p2
    
    -- beam
    local beam = Instance.new("Beam")
    beam.Attachment0 = a1
    beam.Attachment1 = a2
    beam.FaceCamera = true
    beam.Width0 = BEAM_THICKNESS
    beam.Width1 = BEAM_THICKNESS
    beam.LightEmission = 1
    beam.LightInfluence = 0
    beam.TextureSpeed = 2
    beam.TextureLength = 2
    beam.Transparency = NumberSequence.new(0)
    beam.Color = ColorSequence.new(Color3.new(1,1,1))
    beam.Parent = Workspace -- beam can live in Workspace
    
    -- ensure beam is cleaned if either part is removed (safety)
    Debris:AddItem(beam, BEAM_DURATION + 0.5)
    -- attachments will be cleaned with parts (because parent = part)
    
    -- animate colors for finite duration
    animateColorFinite({
    {p1, "partColor"},
    {p2, "partColor"},
    {light1, "lightColor"},
    {light2, "lightColor"},
    {emitter1, "emitterColor"},
    {emitter2, "emitterColor"},
    {beam, "beamColor"}
    })
    
    -- animate movements (finite)
    animateMovement(p2, center2, size2)
    
    -- after BEAM_DURATION ensure destruction (Debris already scheduled), but also explicit cleanup of attachments/beam parent if needed
    task.delay(BEAM_DURATION, function()
        -- explicit safety checks to destroy if still present
        if beam and beam.Parent then
            beam:Destroy()
        end
        -- attachments are children of parts; parts are cleaned by Debris
    end)
end

-- MAIN SAFEGUARDED LOOP
local activeBeams = 0
while true do
    if activeBeams < MAX_BEAMS then
        activeBeams = activeBeams + 1
        task.spawn(function()
            -- spawn one beam pair and automatically animate/clean
            local success, err = pcall(function()
                spawnBeamPair(DROP_CENTER1, DROP_SIZE1, DROP_CENTER2, DROP_SIZE2)
            end)
            if not success then
                warn("Disco spawn error:", err)
            end
            -- ensure activeBeams is decremented after the lifetime plus a small buffer
            task.delay(BEAM_DURATION + 1, function()
                activeBeams = math.max(0, activeBeams - 1)
            end)
        end)
    end
    task.wait(SPAWN_DELAY)
end

-- Animate color for a limited duration (BEAM_DURATION). This uses a finite loop.
local function animateColorFinite(targets)
    -- targets: list of {object, field} pairs. For Beam we set .Color = ColorSequence.new(c)
    local steps = math.max(1, math.floor(BEAM_DURATION / COLOR_STEP))
    task.spawn(function()
        for i = 0, steps do
            local hue = (i / steps)
            local c = rainbowColor(hue)
            for _, t in ipairs(targets) do
                local obj, kind = t[1], t[2]
                if not obj or not obj.Parent then
                    -- skip missing objects
                else
                    if kind == "partColor" then
                        obj.Color = c
                    elseif kind == "lightColor" then
                        obj.Color = c
                    elseif kind == "beamColor" then
                        -- beam.Color expects a ColorSequence
                        if obj and obj.Parent then
                            obj.Color = ColorSequence.new(c)
                        end
                    elseif kind == "emitterColor" then
                        if obj and obj.Parent then
                            obj.Color = ColorSequence.new(c)
                        end
                    end
                end
            end
            task.wait(COLOR_STEP)
        end
    end)
end

-- Create a beam connecting two temporary parts, animate for BEAM_DURATION and ensure cleanup.
local function spawnBeamPair(center1, size1, center2, size2)
    -- create two temporary parts
    local p1, light1, emitter1 = createTempPart(center1, size1)
    local p2, light2, emitter2 = createTempPart(center2, size2)
    
    -- attachments (parented to parts so they are cleaned with parts)
    local a1 = Instance.new("Attachment")
    a1.Parent = p1
    local a2 = Instance.new("Attachment")
    a2.Parent = p2
    
    -- beam
    local beam = Instance.new("Beam")
    beam.Attachment0 = a1
    beam.Attachment1 = a2
    beam.FaceCamera = true
    beam.Width0 = BEAM_THICKNESS
    beam.Width1 = BEAM_THICKNESS
    beam.LightEmission = 1
    beam.LightInfluence = 0
    beam.TextureSpeed = 2
    beam.TextureLength = 2
    beam.Transparency = NumberSequence.new(0)
    beam.Color = ColorSequence.new(Color3.new(1,1,1))
    beam.Parent = Workspace -- beam can live in Workspace
    
    -- ensure beam is cleaned if either part is removed (safety)
    Debris:AddItem(beam, BEAM_DURATION + 0.5)
    -- attachments will be cleaned with parts (because parent = part)
    
    -- animate colors for finite duration
    animateColorFinite({
    {p1, "partColor"},
    {p2, "partColor"},
    {light1, "lightColor"},
    {light2, "lightColor"},
    {emitter1, "emitterColor"},
    {emitter2, "emitterColor"},
    {beam, "beamColor"}
    })
    
    -- animate movements (finite)
    animateMovement(p2, center2, size2)
    
    -- after BEAM_DURATION ensure destruction (Debris already scheduled), but also explicit cleanup of attachments/beam parent if needed
    task.delay(BEAM_DURATION, function()
        -- explicit safety checks to destroy if still present
        if beam and beam.Parent then
            beam:Destroy()
        end
        -- attachments are children of parts; parts are cleaned by Debris
    end)
end

-- MAIN SAFEGUARDED LOOP
local activeBeams = 0
while true do
    if activeBeams < MAX_BEAMS then
        activeBeams = activeBeams + 1
        task.spawn(function()
            -- spawn one beam pair and automatically animate/clean
            local success, err = pcall(function()
                spawnBeamPair(DROP_CENTER1, DROP_SIZE1, DROP_CENTER2, DROP_SIZE2)
            end)
            if not success then
                warn("Disco spawn error:", err)
            end
            -- ensure activeBeams is decremented after the lifetime plus a small buffer
            task.delay(BEAM_DURATION + 1, function()
                activeBeams = math.max(0, activeBeams - 1)
            end)
        end)
    end
    task.wait(SPAWN_DELAY)
end

    -- targets: list of {object, field} pairs. For Beam we set .Color = ColorSequence.new(c)
    local steps = math.max(1, math.floor(BEAM_DURATION / COLOR_STEP))
    task.spawn(function()
        for i = 0, steps do
            local hue = (i / steps)
            local c = rainbowColor(hue)
            for _, t in ipairs(targets) do
                local obj, kind = t[1], t[2]
                if not obj or not obj.Parent then
                    -- skip missing objects
                else
                    if kind == "partColor" then
                        obj.Color = c
                    elseif kind == "lightColor" then
                        obj.Color = c
                    elseif kind == "beamColor" then
                        -- beam.Color expects a ColorSequence
                        if obj and obj.Parent then
                            obj.Color = ColorSequence.new(c)
                        end
                    elseif kind == "emitterColor" then
                        if obj and obj.Parent then
                            obj.Color = ColorSequence.new(c)
                        end
                    end
                end
            end
            task.wait(COLOR_STEP)
        end
    end)
end

-- Create a beam connecting two temporary parts, animate for BEAM_DURATION and ensure cleanup.
local function spawnBeamPair(center1, size1, center2, size2)
    -- create two temporary parts
    local p1, light1, emitter1 = createTempPart(center1, size1)
    local p2, light2, emitter2 = createTempPart(center2, size2)
    
    -- attachments (parented to parts so they are cleaned with parts)
    local a1 = Instance.new("Attachment")
    a1.Parent = p1
    local a2 = Instance.new("Attachment")
    a2.Parent = p2
    
    -- beam
    local beam = Instance.new("Beam")
    beam.Attachment0 = a1
    beam.Attachment1 = a2
    beam.FaceCamera = true
    beam.Width0 = BEAM_THICKNESS
    beam.Width1 = BEAM_THICKNESS
    beam.LightEmission = 1
    beam.LightInfluence = 0
    beam.TextureSpeed = 2
    beam.TextureLength = 2
    beam.Transparency = NumberSequence.new(0)
    beam.Color = ColorSequence.new(Color3.new(1,1,1))
    beam.Parent = Workspace -- beam can live in Workspace
    
    -- ensure beam is cleaned if either part is removed (safety)
    Debris:AddItem(beam, BEAM_DURATION + 0.5)
    -- attachments will be cleaned with parts (because parent = part)
    
    -- animate colors for finite duration
    animateColorFinite({
    {p1, "partColor"},
    {p2, "partColor"},
    {light1, "lightColor"},
    {light2, "lightColor"},
    {emitter1, "emitterColor"},
    {emitter2, "emitterColor"},
    {beam, "beamColor"}
    })
    
    -- animate movements (finite)
    animateMovement(p2, center2, size2)
    
    -- after BEAM_DURATION ensure destruction (Debris already scheduled), but also explicit cleanup of attachments/beam parent if needed
    task.delay(BEAM_DURATION, function()
        -- explicit safety checks to destroy if still present
        if beam and beam.Parent then
            beam:Destroy()
        end
        -- attachments are children of parts; parts are cleaned by Debris
    end)
end

-- MAIN SAFEGUARDED LOOP
local activeBeams = 0
while true do
    if activeBeams < MAX_BEAMS then
        activeBeams = activeBeams + 1
        task.spawn(function()
            -- spawn one beam pair and automatically animate/clean
            local success, err = pcall(function()
                spawnBeamPair(DROP_CENTER1, DROP_SIZE1, DROP_CENTER2, DROP_SIZE2)
            end)
            if not success then
                warn("Disco spawn error:", err)
            end
            -- ensure activeBeams is decremented after the lifetime plus a small buffer
            task.delay(BEAM_DURATION + 1, function()
                activeBeams = math.max(0, activeBeams - 1)
            end)
        end)
    end
    task.wait(SPAWN_DELAY)
end

-- helper: safe random position in a rectangle (keeps Y fixed)
local function getRandomDropPosition(center, size)
    local minX, maxX = center.X - size.X/2, center.X + size.X/2
    local minZ, maxZ = center.Z - size.Z/2, center.Z + size.Z/2
    return Vector3.new(
    math.random() * (maxX - minX) + minX,
    center.Y,
    math.random() * (maxZ - minZ) + minZ
    )
end
 
local function rainbowColor(h)
    return Color3.fromHSV(h % 1, 1, 1)
end
 
-- Creates a temporary disco part (returns part)
local function createTempPart(center, size)
    local p = Instance.new("Part")
    p.Anchored = true
    p.CanCollide = false
    p.Material = Enum.Material.Neon
    p.Shape = Enum.PartType.Ball
    p.Transparency = 1
    p.Size = Vector3.new(2,2,2)
    p.Position = getRandomDropPosition(center, size)
    p.Parent = fol
    
    -- point light
    local light = Instance.new("PointLight")
    light.Brightness = 6
    light.Range = 25
    light.Parent = p
    
    -- particle emitter (attached to part so Debris cleans it automatically)
    local emitter = Instance.new("ParticleEmitter")
    emitter.Texture = "rbxassetid://619439195"
    emitter.Rate = 35
    emitter.Lifetime = NumberRange.new(0.3, 0.6)
    emitter.Speed = NumberRange.new(0.5, 1.5)
    emitter.Size = NumberSequence.new({NumberSequenceKeypoint.new(0,0.4), NumberSequenceKeypoint.new(1,0)})
    emitter.LightEmission = 1
    emitter.Parent = p
    
    -- schedule automatic cleanup via Debris (safest)
    Debris:AddItem(p, BEAM_DURATION + 1)
    
    return p, light, emitter
end
 
-- Move part smoothly a few times while it exists, but stop when it is removed.
-- We animate movement using a finite number of steps rather than an infinite loop.
local function animateMovement(part, center, size)
    -- number of moves during lifetime
    local moves = math.max(1, math.floor((BEAM_DURATION)/math.max(0.0001, MOVE_INTERVAL)))
    task.spawn(function()
        for i = 1, moves do
            if not part or not part.Parent then break end
            local newPos = getRandomDropPosition(center, size)
            local tween = TweenService:Create(part, TweenInfo.new(MOVE_SPEED, Enum.EasingStyle.Sine, Enum.EasingDirection.InOut), {Position = newPos})
            tween:Play()
            -- wait smaller increments so if the part is removed we break quickly
            local waited = 0
            while waited < MOVE_INTERVAL do
                if not part or not part.Parent then break end
                task.wait(0.1)
                waited = waited + 0.1
            end
            if not part or not part.Parent then break end
        end
    end)
end

-- Animate color for a limited duration (BEAM_DURATION). This uses a finite loop.

            local c = rainbowColor(hue)
            for _, t in ipairs(targets) do
                local obj, kind = t[1], t[2]
                if not obj or not obj.Parent then
                    -- skip missing objects
                else
                    if kind == "partColor" then
                        obj.Color = c
                    elseif kind == "lightColor" then
                        obj.Color = c

                        -- beam.Color expects a ColorSequence
                        if obj and obj.Parent then
                            obj.Color = ColorSequence.new(c)
                        end
                    elseif kind == "emitterColor" then
                        if obj and obj.Parent then
                            obj.Color = ColorSequence.new(c)

            task.wait(COLOR_STEP)

    beam.Width1 = BEAM_THICKNESS
    beam.LightEmission = 1
    beam.LightInfluence = 0
    beam.TextureSpeed = 2
    beam.TextureLength = 2
    beam.Transparency = NumberSequence.new(0)
    beam.Color = ColorSequence.new(Color3.new(1,1,1))
    beam.Parent = Workspace -- beam can live in Workspace
    
    -- ensure beam is cleaned if either part is removed (safety)
    Debris:AddItem(beam, BEAM_DURATION + 0.5)
    -- attachments will be cleaned with parts (because parent = part)
    
    -- animate colors for finite duration
    animateColorFinite({
    {p1, "partColor"},
    {p2, "partColor"},

        if beam and beam.Parent then

    end)

            -- ensure activeBeams is decremented after the lifetime plus a small buffer
            task.delay(BEAM_DURATION + 1, function()
                activeBeams = math.max(0, activeBeams - 1)
            end)
        end)
    end
    task.wait(SPAWN_DELAY)
end
